# 1: open myphpadmin
# 2: create database "general_ledger"
# 3: import the "general_ledger.sql"
# 4: go to vs code terminal in bash environment
# 5: type the command 'pip install virtualenv'
# 6: type the command 'virtualenv venv'
# 7: type the command 'source venv/Scripts/activate'
# 8: type the command 'pip install -r requirements.txt' OR 'pip list --format=freeze > requirements.txt'
# 9: type the command 'uvicorn homies.main:app --reload'
# 10: go to http://127.0.0.1:8000/generalledger/
